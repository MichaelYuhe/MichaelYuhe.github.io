var longestConsecutive = function(nums) {
    res = 0
    let myHash = new Set(nums)
    for(let num of myHash) {
        if(!myHash.has(num - 1)) {
            let currentNum = num
            let currentLen = 1
            while(myHash.has(currentNum + 1)) {
                currentNum++
                currentLen++
            }
            res = Math.max(res, currentLen)
        }
    }
    return res
};
